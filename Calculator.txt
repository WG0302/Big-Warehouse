using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace Calculator
{
    class Object
    {

        public static bool Equals(Object a, Object b)
        {
            if (a == b)
            {
                return true;
                
            }
            if (a == null || b == null)
            {
                return false;
            }
            return a.Equals(b); 
        }
    }
}
